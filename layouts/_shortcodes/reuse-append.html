Example: {{< reuse-append "reuse/table-base.md" >}} | Extra Row | Value | {{< /reuse-append >}} */ -}} {{- $asset := "" -}} {{- with (.Get 0) -}} {{- $asset = . -}} {{- else -}} {{- errorf "The %q shortcode requires one positional parameter: path in /assets. See %s" .Name .Position -}} {{- end -}} {{- $res := resources.Get $asset -}} {{- if not $res -}} {{- errorf "The %q shortcode could not find %q. See %s" .Name $asset .Position -}} {{- end -}} {{- $base := trim $res.Content " \n\r\t" -}} {{- $extra := trim .Inner " \n\r\t" -}} {{- $combined := printf "%s\n%s\n" $base $extra -}} {{- $combined | .Page.RenderString -}}
