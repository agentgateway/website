{{- /* Shortcode: link-hextra Generates a section-aware relative URL by prepending the current page's section prefix (up to 3 segments deep) to the provided path. Parameters: - path: The path to append to the current section prefix (e.g., "/getting-started/install/"). Example: {{< link-hextra path="/getting-started/install/" >}} */ -}} {{- $path := .Get "path" -}} {{- $rel := .Page.RelPermalink -}} {{- $match3 := findRE `^/[^/]+/[^/]+/[^/]+` $rel -}} {{- $match2 := findRE `^/[^/]+/[^/]+` $rel -}} {{- $prefix := "" -}} {{- $product := "" -}} {{- if $match3 -}} {{- $prefix = index $match3 0 -}} {{- $segments := split $prefix "/" -}} {{- $product = index $segments 2 -}} {{- else if $match2 -}} {{- $prefix = index $match2 0 -}} {{- end -}} {{- if and $product (hasPrefix $path (printf "/%s" $product)) -}} {{- $path = strings.TrimPrefix (printf "/%s" $product) $path -}} {{- end -}} {{- $prefix }}{{ $path -}}
