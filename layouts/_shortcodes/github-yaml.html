{{- /* Shortcode: github-yaml Fetches a remote YAML file by URL and renders the content in a syntax-highlighted YAML code block, stripping any yaml-language-server directives and refreshing the cache daily. Parameters: - url: The remote URL of the YAML file to fetch and display. Example: {{< github-yaml url="https://raw.githubusercontent.com/org/repo/main/config.yaml" >}} On the SUCCESS path nothing extra is emitted — just the code block. On FAILURE (dead/moved URL) the error message renders the URL as a real . That is deliberate: the link checker scans rendered HTML and follows links, so a broken include surfaces as a reported dead link instead of silent error text. CI builds cache-cold, so a URL that is dead at build time always lands on this branch. The standalone error block is not inside a 
, so lychee extracts the
  anchor (anchors inside code blocks are skipped by default).
*/ -}}
{{- $url := .Get "url" -}}
{{- /* Use daily cache key to ensure content refreshes at least once per day */ -}}
{{- $cacheKey := printf "%s-%s" $url (now.Format "2006-01-02") -}}
{{- with resources.GetRemote $url (dict "key" $cacheKey) -}}
```yaml {base_url="{{ $url | path.Dir}}",filename="{{ $url | path.Base}}"}
{{ .Content | replaceRE "^# yaml-language-server.*\n" "" | safe.HTML}}
```

{{- else -}}
  
Error: Unable to load code from {{ $url }}

{{- end -}}
