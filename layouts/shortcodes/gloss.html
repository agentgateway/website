{{- /* Shortcode: gloss Renders an inline glossary term with a native tooltip sourced from data/{lang}/termbase. Parameters: - 0 (positional): Glossary key — matches abbr or term (case-insensitive). - (inner content): Optional custom display text; defaults to the key if omitted. Example: {{< gloss "MCP" >}}custom display text{{< /gloss >}} */ -}} {{- $key := .Get 0 -}} {{- $keyLower := lower $key -}} {{- $match := dict -}} {{- range (index .Site.Data .Site.Language.Lang "termbase") -}} {{- if or (eq (lower .abbr) $keyLower) (eq (lower .term) $keyLower) -}} {{- $match = . -}} {{- break -}} {{- end -}} {{- end -}} {{- $display := .Inner | default $key -}} {{- with $match -}}{{ $display }}{{- else -}}{{ $display }}{{- end -}}
