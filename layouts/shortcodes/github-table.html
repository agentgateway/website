{{- /* Shortcode: github-table Fetches a remote Markdown file by URL and extracts a specific section by heading, optionally excluding lines that match a regex pattern. Parameters: - url: The remote URL of the Markdown file to fetch. - section: The heading name of the section to extract. - exclude: (optional) A regex pattern for lines to remove from the extracted section. Example: {{< github-table url="https://raw.githubusercontent.com/org/repo/main/README.md" section="Compatibility" exclude="^\\|.*draft.*" >}} */ -}} {{- $url := .Get "url" -}} {{ $exclude := .Get "exclude" }} {{- $section := .Get "section" -}} {{- with resources.GetRemote $url -}} {{- $content := .Content -}} {{- $pattern := printf "(?ms)^#+? %s\\s*(.*?)(^#+? |\\z)" $section -}} {{- $match := index (findRESubmatch $pattern $content) 0 -}} {{- if $match -}} {{ $content := index $match 1 | replaceRE (printf "^#+? %s\\s*" $section) "" -}} {{ if $exclude -}} {{ replaceRE (printf "(?m)%s.*\n?" $exclude) "" $content |safe.HTML }} {{ else -}} {{ $content|safe.HTML }} {{ end }} {{- else -}} {{ errorf "fail to find section: %s" $section }} {{- end -}} {{- else -}} {{ errorf "unable to load content from %s" $url }} {{- end -}}
