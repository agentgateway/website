{{- /* Copy as Markdown button — uses Hugo's transform.HTMLToMarkdown (v0.151+) to convert rendered HTML back to clean markdown for clipboard copy. */ -}} {{- $html := .Content -}} {{- /* Convert Hextra card links to simple

links before markdown conversion, otherwise transform.HTMLToMarkdown drops them entirely. Matches: Title Also: ...Title... */ -}} {{- $html = $html | replaceRE `(?s)]*href="([^"]*)"[^>]*>.*?]*>([^<]+).*?` "

$2

" -}} {{- /* Remove card grid wrappers and their inline styles */ -}} {{- $html = $html | replaceRE `(?s)
]*>(.*?)
` "$1" -}} {{- $html = $html | replaceRE `(?s)` "" -}} {{- /* Chroma linenos=table: replace the two-cell table with just the code-side 
.
       Otherwise HTMLToMarkdown serializes the line-number  as its own fenced block
       above the actual code, producing an orphan "1\n2\n..." block in the copy output. */ -}}
{{- $html = $html | replaceRE `(?s)\s*\s*\s*
\s*]*>.*?\s*(]*>]*>.*?)\s*
` "$1" -}}
{{- /* Tabs: strip the button bar, unhide all panels, and inject a label before each panel
       using the aria-labelledby to match the tab button's text. */ -}}
{{- /* Remove entire tablist div (buttons) — we'll label panels directly */ -}}
{{- $html = $html | replaceRE `(?s)]*role="tablist"[^>]*>.*?` "" -}}
{{- /* Unhide all panels and inject "Option:" label using the panel's aria-labelledby ID to
       extract the tab name from the corresponding button. Since buttons are gone, we use
       the tab ID suffix as a fallback — but simpler: just add a marker we'll fix in JS. */ -}}
{{- $html = $html | replaceRE `aria-hidden="true"` "" -}}
{{- $html = $html | replaceRE `
]*aria-labelledby="([^"]*)"[^>]*>` `
Option: $1
` -}}
{{- $md := printf "# %s\n\n%s" .Title ($html | transform.HTMLToMarkdown) -}}
{{- /* Strip empty heading-permalink anchors only — [](#id) → "".
       Preserve [text](#id) cross-section links so readers retain the reference. */ -}}
{{- $md = $md | replaceRE `\[\]\(#[^)]+\)` "" -}}
{{- /* Normalize Goldmark/typographer-introduced smart punctuation back to ASCII so the
       copied markdown survives plain-text contexts (LLM prompts, terminals, code blocks). */ -}}
{{- $md = $md | replaceRE "[“”]" "\"" -}}
{{- $md = $md | replaceRE "[‘’]" "'" -}}
{{- $md = $md | replaceRE "…" "..." -}}
{{- $md = $md | replaceRE "–" "-" -}}
{{- $md = $md | replaceRE "—" "--" -}}
{{- /* Resolve a canonical base URL for the copied markdown so links resolve when pasted
       elsewhere. site.BaseURL is "/" in this repo (root-relative output), so fall back to
       the production hostname when it lacks a scheme/host. */ -}}
{{- $siteBase := site.BaseURL | strings.TrimRight "/" -}}
{{- $baseURL := $siteBase -}}
{{- if not (or (hasPrefix $siteBase "http://") (hasPrefix $siteBase "https://")) -}}
  {{- $baseURL = "https://agentgateway.dev" -}}
{{- end -}}
{{- /* Current page URL (without any trailing slash), normalized to use the canonical baseURL.
       In dev .Permalink is //localhost:PORT/path/, in prod it's https://host/path/. */ -}}
{{- $pagePath := .Permalink | replaceRE `^(https?:)?//[^/]+` "" -}}
{{- $pageAbs := printf "%s%s" $baseURL $pagePath | strings.TrimRight "/" -}}

{{- /* Image paths → absolute URLs.
       Use ${1}/${2} (not $1/$2) so Go's regex engine doesn't merge the digit with following
       letters into an undefined named group — e.g. "$1https" parses as name "1https" → "". */ -}}
{{- $md = $md | replaceRE `(!\[[^\]]*\]\()//localhost:\d+/` (printf "${1}%s/" $baseURL) -}}
{{- $md = $md | replaceRE `(!\[[^\]]*\]\()(/[^)]+)` (printf "${1}%s${2}" $baseURL) -}}
{{- /* Fill empty image alt text with filename — ![](/path/foo-bar.png) → ![foo-bar](/path/foo-bar.png) */ -}}
{{- $md = $md | replaceRE `!\[\]\(([^)]*/)([^/)]+\.[a-zA-Z]+)\)` "![${2}](${1}${2})" -}}

{{- /* Link URLs → absolute URLs. Run AFTER image rewriting so already-https image URLs
       don't match the relative-link pattern below. */ -}}
{{- /* Localhost-rooted links → baseURL/path */ -}}
{{- $md = $md | replaceRE `(\[[^\]]+\]\()//localhost:\d+/` (printf "${1}%s/" $baseURL) -}}
{{- /* Root-relative links (/docs/…) → baseURL/path. Excludes anchors via [^)#]. */ -}}
{{- $md = $md | replaceRE `(\[[^\]]+\]\()(/[^)#]+)` (printf "${1}%s${2}" $baseURL) -}}
{{- /* Relative links (basic, basic/sub, basic#anchor) → pageAbs/relative. First URL char
       must be alphanum or _/-; subsequent chars exclude ) and : so we skip absolute URLs
       (https://…, mailto:…) but allow # for fragment-suffixed relative links. */ -}}
{{- $md = $md | replaceRE `(\[[^\]]+\]\()([a-zA-Z0-9_\-][^):]*)(\))` (printf "${1}%s/${2}${3}" $pageAbs) -}}
