"" -}} {{- $pathSegments = $pathSegments | append . -}} {{- end -}} {{- end -}} {{- $isVersionIndex := eq (len $pathSegments) 0 -}} {{- /* Build breadcrumb items for paths after version */ -}} {{- $breadcrumbItems := slice -}} {{- if not $isVersionIndex -}} {{- $basePath := printf "/docs/%s/%s/" $subSection $currentVersion -}} {{- $currentPath := $basePath -}} {{- range $index, $segment := $pathSegments -}} {{- $currentPath = printf "%s%s/" $currentPath $segment -}} {{- $page := $.Site.GetPage $currentPath -}} {{- if $page -}} {{- $title := $page.Title | default $page.LinkTitle | default $segment -}} {{- $isLast := eq $index (sub (len $pathSegments) 1) -}} {{- $breadcrumbItems = $breadcrumbItems | append (dict "title" $title "url" $currentPath "isLast" $isLast) -}} {{- end -}} {{- end -}} {{- end -}}
